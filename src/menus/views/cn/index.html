菜单

实现统一的菜单组件，保证风格统一

默认菜单 
html:
<oasis-menu  opts="opts" ></oasis-menu>
js:
$scope.opts = {
  data:[
    {
      "icon":"glyphicon glyphicon-education",
      "name":"菜单一",
      "sref":"a",
      "child":[
      {
        "icon" :"glyphicon glyphicon-education",
        "name":"菜单二",
        "sref":"b",
        "child":[
        {
          "icon": "glyphicon glyphicon-education",
          "sref": "f",
          "name":"菜单三" 
        }] 
      }]  
    }
  ]
};
